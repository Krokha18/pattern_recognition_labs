Застосувати алгоритм персептрона для класифікації цифр¶

In [12]:
def train_perceptron(X_t, y_t):
    alphas = np.zeros((10,28,28))
    bs = np.zeros((10,))
    it=1
    while(True):
        X_t, y_t  = shuffle(X_t, y_t)
        for x, y in zip(X_t, y_t):
            predict = (alphas*x).sum(axis=(1,2)) + bs
            for ind in range(10):
                if(ind!=y and predict[ind]>0):
                    alphas[ind] -= x
                    bs[ind] -=1
                if(ind==y and predict[ind]<=0):
                    alphas[ind] += x
                    bs[ind] +=1
        predicts = np.array( [np.argmax( (alphas*x).sum(axis=(1,2))+ bs ) for x in X_t] )
        n_wrong = (predicts!=y_t).sum()
        print(f"it = {it}, n_wrong={n_wrong}")
        it+=1
        if(n_wrong==0):
            return alphas, bs
